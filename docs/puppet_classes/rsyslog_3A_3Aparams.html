class rsyslog::params {
  $service_name       = 'rsyslog'
  if ($facts['os']['name'] in ['RedHat','CentOS']) and ($facts['os']['release']['major'] == '6') {
    $package_name  = 'rsyslog7'
    $read_journald = false
  }
  else {
    $package_name  = 'rsyslog'
    $read_journald = true
  }

  $tls_package_name = "${package_name}-gnutls"
}
